
## Part 1: Coroutine 跨平台原理 & 实现

### 核心问题

Kotlin Coroutine 在不同平台有不同的线程模型：

- **Android**：Main Thread + ThreadPool
- **iOS**：GCD (Grand Central Dispatch) 调度
- **JVM**：ForkJoinPool

**关键是：suspend 函数需要在合适的线程上执行！**

```kotlin
fun CoroutineScope.launchSafe(
    onError: (Throwable) -> Unit = {},
    block: suspend () -> Unit
) {
    launch(CoroutineExceptionHandler { _, exception ->
        onError(exception)
    }) {
        block()
    }
}
```

## Part 2: Flow 跨平台状态管理

### 通用 ViewModel 设计（commonMain）

```kotlin
// shared/src/commonMain/kotlin/viewmodel/UserViewModel.kt
import kotlinx.coroutines.*
import kotlinx.coroutines.flow.*

class UserViewModel {
    private val _userState = MutableStateFlow<UserUIState>(UserUIState.Loading)
    val userState: StateFlow<UserUIState> = _userState.asStateFlow()
    
    private val _errorFlow = MutableSharedFlow<String>()
    val errorFlow: SharedFlow<String> = _errorFlow.asSharedFlow()
    
    private val scope = CoroutineScopeProvider().createMainScope()
    
    fun loadUser(userId: String) {
        scope.launch {
            try {
                _userState.value = UserUIState.Loading
                
                // 在 IO 线程获取数据
                val user = withContext(ioDispatcher) {
                    UserRepository.getUser(userId)
                }
                
                // 回到 Main 线程更新 UI 状态
                _userState.value = UserUIState.Success(user)
            } catch (e: Exception) {
                _userState.value = UserUIState.Error(e.message ?: "Unknown error")
                _errorFlow.emit("加载失败: ${e.message}")
            }
        }
    }
    
    fun destroy() {
        scope.cancel()
    }
}

sealed class UserUIState {
    data object Loading : UserUIState()
    data class Success(val user: User) : UserUIState()
    data class Error(val message: String) : UserUIState()
}
```

```swift
// iosApp/ViewController.swift
import SwiftUI

class IosUserViewModel: ObservableObject {

    private let kmpViewModel = UserViewModel()
    private var job: Job?
    
    func loadUser(userId: String) {

        job = kmpViewModel.userState.collect { [weak self] state in

        job?.cancel()
        kmpViewModel.destroy()

                EmptyView()
            }
        }

        }
```

### 2️⃣ commonMain - 业务层实现

```kotlin
// shared/src/commonMain/kotlin/network/ApiService.kt
class ApiService(private val client: NetworkClient) {
    
    // 日志拦截器
    private val loggingInterceptor = object : HttpInterceptor {
        override suspend fun intercept(chain: HttpChain): HttpResponse {
            val request = chain.request
            val startTime = currentTimeMillis()
            
            logger.debug("→ ${request.method} ${request.url}")
            request.headers.forEach { (k, v) ->
                logger.debug("  $k: $v")
            }
            
            return try {
                val response = chain.proceed(request)
                val duration = currentTimeMillis() - startTime
                logger.debug("← ${response.code} (耗时 ${duration}ms)")
                response
            } catch (e: Exception) {
                logger.error("✗ 请求失败", e)
                throw e
            }
        }
    }
    
    suspend inline fun <reified T> getUser(userId: String): User {
        return try {
            val response = client.get<ApiResponse<User>>(
                path = "/api/users/$userId",
                headers = mapOf("Authorization" to "Bearer ${getToken()}")
            )
            
            if (response.code == 200 && response.data != null) {
                response.data
            } else {
                throw HttpException(
                    response.code,
                    response.message
                )
            }
        } catch (e: HttpException) {
            logger.error("获取用户失败: ${e.message}")
            throw e
        }
    }
    
    suspend fun createUser(user: User): User {
        return client.post<User>(
            path = "/api/users",
            body = user,
            headers = mapOf("Authorization" to "Bearer ${getToken()}")
        ).data ?: throw HttpException(500, "Server returned null")
    }
    
    private suspend fun getToken(): String {
        // 从本地存储获取 token
        return PlatformStorage().read("auth_token") ?: ""
    }
}
```

### 3️⃣ androidMain - Ktor 实现

```kotlin
// shared/src/androidMain/kotlin/network/HttpClient.kt
import io.ktor.client.*
import io.ktor.client.features.*
import io.ktor.client.serialization.*
import io.ktor.client.engine.okhttp.*
import kotlinx.serialization.json.Json
import okhttp3.logging.HttpLoggingInterceptor

actual val httpClient: NetworkClient by lazy {
    KtorNetworkClient(createKtorHttpClient())
}

fun createKtorHttpClient(): HttpClient {
    return HttpClient(OkHttp) {
        // OkHttp 日志拦截
        engine {
            addInterceptor(HttpLoggingInterceptor().apply {
                level = HttpLoggingInterceptor.Level.BODY
            })
            
            // 超时配置
            connectTimeout = 30_000
            readTimeout = 30_000
            writeTimeout = 30_000
        }
        
        // JSON 序列化
        install(JsonFeature) {
            serializer = KotlinxSerializer(Json {
                ignoreUnknownKeys = true
                encodeDefaults = true
            })
        }
        
        // 异常处理
        install(HttpTimeout) {
            requestTimeoutMillis = 30_000
            socketTimeoutMillis = 30_000
        }
    }
}

actual class HttpClientBuilder {
    private val client = createKtorHttpClient()
    
    actual fun addInterceptor(interceptor: HttpInterceptor): HttpClientBuilder {
        // Ktor 的拦截器集成
        return this
    }
    
    actual fun setTimeout(timeoutMs: Long): HttpClientBuilder {
        return this
    }
    
    actual fun build(): NetworkClient = KtorNetworkClient(client)
}

class KtorNetworkClient(private val httpClient: HttpClient) : NetworkClient {
    
    override suspend fun <T> get(
        path: String,
        headers: Map<String, String>
    ): ApiResponse<T> = withContext(ioDispatcher) {
        try {
            httpClient.get(path) {
                headers.forEach { (k, v) -> header(k, v) }
            }
        } catch (e: Exception) {
            throw HttpException(500, e.message ?: "Network error", e)
        }
    }
    
    override suspend fun <T> post(
        path: String,
        body: Any,
        headers: Map<String, String>
    ): ApiResponse<T> = withContext(ioDispatcher) {
        try {
            httpClient.post(path) {
                headers.forEach { (k, v) -> header(k, v) }
                this.body = body
            }
        } catch (e: Exception) {
            throw HttpException(500, e.message ?: "Network error", e)
        }
    }
    
    override suspend fun <T> execute(
        path: String,
        method: String,
        body: Any?,
        headers: Map<String, String>
    ): ApiResponse<T> {
        return when (method) {
            "GET" -> get(path, headers)
            "POST" -> post(path, body ?: Unit, headers)
            else -> throw HttpException(400, "Unsupported method: $method")
        }
    }
}
```

### 4️⃣ iosMain - Alamofire 实现

```kotlin
// shared/src/iosMain/kotlin/network/HttpClient.kt
import platform.Foundation.*
import kotlinx.coroutines.*
import platform.Darwin.dispatch_get_main_queue
import platform.Darwin.dispatch_async

actual val httpClient: NetworkClient by lazy {
    IosNetworkClient()
}

actual class HttpClientBuilder {
    actual fun addInterceptor(interceptor: HttpInterceptor): HttpClientBuilder = this
    actual fun setTimeout(timeoutMs: Long): HttpClientBuilder = this
    actual fun build(): NetworkClient = IosNetworkClient()
}

class IosNetworkClient : NetworkClient {
    
    override suspend fun <T> get(
        path: String,
        headers: Map<String, String>
    ): ApiResponse<T> = withContext(ioDispatcher) {
        performRequest(path, "GET", null, headers)
    }
    
    override suspend fun <T> post(
        path: String,
        body: Any,
        headers: Map<String, String>
    ): ApiResponse<T> = withContext(ioDispatcher) {
        val jsonData = Json.encodeToString(body)
        performRequest(path, "POST", jsonData, headers)
    }
    
    override suspend fun <T> execute(
        path: String,
        method: String,
        body: Any?,
        headers: Map<String, String>
    ): ApiResponse<T> {
        return when (method) {
            "GET" -> get(path, headers)
            "POST" -> post(path, body ?: Unit, headers)
            else -> throw HttpException(400, "Unsupported method: $method")
        }
    }
    
    private suspend inline fun <reified T> performRequest(
        path: String,
        method: String,
        body: String?,
        headers: Map<String, String>
    ): ApiResponse<T> = suspendCancellableCoroutine { continuation ->
        val url = NSURLComponents().apply {
            this.scheme = "https"
            this.host = "api.example.com"
            this.path = path
        }.URL ?: run {
            continuation.resumeWithException(HttpException(400, "Invalid URL"))
            return@suspendCancellableCoroutine
        }
        
        val request = NSMutableURLRequest(url).apply {
            HTTPMethod = method
            headers.forEach { (k, v) ->
                setValue(v, forHTTPHeaderField = k)
            }
            if (body != null) {
                HTTPBody = body.encodeToByteArray().toNSData()
            }
        }
        
        NSURLSession.sharedSession.dataTaskWithRequest(request) { data, response, error ->
            when {
                error != null -> {
                    continuation.resumeWithException(
                        HttpException(500, error.description, Exception(error.description))
                    )
                }
                data != null -> {
                    try {
                        val jsonString = NSString(data, NSUTF8StringEncoding)
                        val result = Json.decodeFromString<ApiResponse<T>>(jsonString)
                        continuation.resume(result)
                    } catch (e: Exception) {
                        continuation.resumeWithException(
                            HttpException(500, "Parse error", e)
                        )
                    }
                }
                else -> {
                    continuation.resumeWithException(
                        HttpException(500, "Empty response")
                    )
                }
            }
        }.resume()
    }
}
```

## Part 4: 完整的使用示例

### commonMain 仓储层

```kotlin
// shared/src/commonMain/kotlin/repository/UserRepository.kt
class UserRepository(
    private val apiService: ApiService,
    private val storage: PlatformStorage
) {
    
    suspend fun fetchUser(userId: String): User {
        return try {
            logger.debug("从网络获取用户 $userId")
            val user = apiService.getUser(userId)
            
            // 缓存到本地
            storage.save("user_$userId", Json.encodeToString(user))
            user
        } catch (e: HttpException) {
            logger.warn("网络请求失败，尝试读取本地缓存")
            
            // 缓存降级
            val cached = storage.read("user_$userId")
            if (cached != null) {
                Json.decodeFromString<User>(cached)
            } else {
                throw e
            }
        }
    }
    
    suspend fun watchUserUpdates(userId: String): Flow<User> = flow {
        while (isActive) {
            try {
                val user = fetchUser(userId)
                emit(user)
                delay(5000) // 每 5 秒轮询一次
            } catch (e: Exception) {
                logger.error("轮询失败", e)
                delay(5000)
            }
        }
    }
}

// ViewModel 中使用
class UserDetailViewModel(
    private val repository: UserRepository
) : ViewModel() {
    
    private val _userFlow = MutableStateFlow<User?>(null)
    val userFlow: StateFlow<User?> = _userFlow.asStateFlow()
    
    fun startWatching(userId: String) {
        viewModelScope.launch {
            repository.watchUserUpdates(userId)
                .collect { user ->
                    _userFlow.value = user
                }
        }
    }
}
```

## 面试重点总结

|概念|关键点|
|---|---|
|**Dispatcher 跨平台**|mainDispatcher 和 ioDispatcher 的平台适配|
|**StateFlow vs SharedFlow**|StateFlow 有状态缓冲，SharedFlow 用于事件分发|
|**suspend 函数**|编译器生成 continuation 对象，支持 callback 回调|
|**Ktor vs OkHttp**|Ktor 多平台原生支持，OkHttp 仅 JVM/Android|
|**拦截器链**|Chain of Responsibility 模式，支持日志/认证/重试|
|**缓存降级**|网络异常时读取本地存储，提升稳定性|
|**Flow 背压**|collect 是挂起函数，自动处理背压|

这套网络架构在 **Kingdee Cloud 和智能座舱项目中都是标准模式**。你现在能理解为什么这样设计吗？
