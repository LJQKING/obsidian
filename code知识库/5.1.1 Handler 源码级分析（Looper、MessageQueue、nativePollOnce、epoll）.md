## 一、`MessageQueue.next()` — Java 层入口

```java
// frameworks/base/core/java/android/os/MessageQueue.java

Message next() {
    final long ptr = mPtr;          // NativeMessageQueue* 指针
    if (ptr == 0) return null;      // 队列已销毁

    int pendingIdleHandlerCount = -1;
    int nextPollTimeoutMillis = 0;  // 0 = 立即返回；-1 = 永久阻塞

    for (;;) {
        if (nextPollTimeoutMillis != 0) {
            Binder.flushPendingCommands(); // 刷新 Binder 事务再睡眠
        }

        // ★ 核心：跨越 JNI 进入 native，可能在此阻塞任意长时间
        nativePollOnce(ptr, nextPollTimeoutMillis);

        synchronized (this) {
            final long now = SystemClock.uptimeMillis();
            Message prevMsg = null;
            Message msg = mMessages;

            // 遇到同步屏障（target==null），跳过同步消息找异步消息
            if (msg != null && msg.target == null) {
                do {
                    prevMsg = msg;
                    msg = msg.next;
                } while (msg != null && !msg.isAsynchronous());
            }

            if (msg != null) {
                if (now < msg.when) {
                    // 未到期：精确计算下次超时，最大值 Integer.MAX_VALUE
                    nextPollTimeoutMillis = (int) Math.min(
                        msg.when - now, Integer.MAX_VALUE);
                } else {
                    // 到期：摘链、返回
                    mBlocked = false;
                    if (prevMsg != null) prevMsg.next = msg.next;
                    else mMessages = msg.next;
                    msg.next = null;
                    msg.markInUse();
                    return msg;                // ← 唯一的正常出口
                }
            } else {
                nextPollTimeoutMillis = -1;    // 队列空，永久阻塞
            }

            // 队列空闲时执行 IdleHandler
            if (pendingIdleHandlerCount < 0
                    && (mMessages == null || now < mMessages.when)) {
                pendingIdleHandlerCount = mIdleHandlers.size();
            }
            // ... 执行 IdleHandler，执行完后下一轮 nextPollTimeoutMillis=0
        }
    }
}
```

## 二、JNI 桥接层

`nativePollOnce` 是一个 `native` 方法，声明与注册：

```cpp
// frameworks/base/core/jni/android_os_MessageQueue.cpp

// Java 侧声明
// private native void nativePollOnce(long ptr, int timeoutMillis);

static void android_os_MessageQueue_nativePollOnce(
        JNIEnv* env, jobject obj, jlong ptr, jint timeoutMillis) {

    // ptr 就是 new NativeMessageQueue() 的裸指针，强转回来
    NativeMessageQueue* nativeMessageQueue =
        reinterpret_cast<NativeMessageQueue*>(ptr);

    nativeMessageQueue->pollOnce(env, obj, timeoutMillis);
}

// NativeMessageQueue 只是薄薄的一层包装
void NativeMessageQueue::pollOnce(JNIEnv* env, jobject pollObj, int timeoutMillis) {
    mPollEnv = env;
    mPollObj = pollObj;

    // 实际工作全在 C++ Looper 里
    mLooper->pollOnce(timeoutMillis);

    mPollObj = NULL;
    mPollEnv = NULL;
}

// 唤醒侧同理
static void android_os_MessageQueue_nativeWake(
        JNIEnv* env, jobject obj, jlong ptr) {
    NativeMessageQueue* nativeMessageQueue =
        reinterpret_cast<NativeMessageQueue*>(ptr);
    nativeMessageQueue->wake();
}

void NativeMessageQueue::wake() {
    mLooper->wake();
}
```

## 三、C++ `Looper` — epoll 的真正操作者

`Looper` 的初始化是整个机制的基础：

```cpp
// system/core/libutils/Looper.cpp

Looper::Looper(bool allowNonCallbacks)
    : mAllowNonCallbacks(allowNonCallbacks),
      mSendingMessage(false),
      mPolling(false),
      mEpollRebuildRequired(false),
      mNextRequestSeq(0),
      mResponseIndex(0),
      mNextMessageUptime(LLONG_MAX) {

    // ① 创建 eventfd（值为 0，每次 write +1，read 清零）
    mWakeEventFd = eventfd(0, EFD_NONBLOCK | EFD_CLOEXEC);

    // ② 创建 epoll 实例
    mEpollFd = epoll_create1(EPOLL_CLOEXEC);

    // ③ 把 mWakeEventFd 注册到 epoll，监听"可读"事件
    struct epoll_event wakeEvent;
    wakeEvent.events = EPOLLIN;         // 只关心可读
    wakeEvent.data.fd = mWakeEventFd;
    epoll_ctl(mEpollFd, EPOLL_CTL_ADD, mWakeEventFd, &wakeEvent);
}
```

### `pollInner` — 核心阻塞逻辑

```cpp
int Looper::pollInner(int timeoutMillis) {
    // 计算精确超时（考虑 native 消息队列里的定时任务）
    if (timeoutMillis != 0 && mNextMessageUptime != LLONG_MAX) {
        nsecs_t now = systemTime(SYSTEM_TIME_MONOTONIC);
        int messageTimeoutMillis = toMillisecondTimeoutDelay(now, mNextMessageUptime);
        if (messageTimeoutMillis >= 0
                && (timeoutMillis < 0 || messageTimeoutMillis < timeoutMillis)) {
            timeoutMillis = messageTimeoutMillis;
        }
    }

    struct epoll_event eventItems[EPOLL_MAX_EVENTS];  // 最多 16 个事件

    // ★★★ 核心系统调用：线程在此真正睡眠 ★★★
    // timeoutMillis == -1 → 永久等待
    // timeoutMillis ==  0 → 立即返回（poll 模式）
    // timeoutMillis >   0 → 最多等待 N 毫秒
    int eventCount = epoll_wait(mEpollFd, eventItems, EPOLL_MAX_EVENTS, timeoutMillis);

    // ---- 从此往下：线程已被唤醒 ----

    if (eventCount < 0) {
        // EINTR：被信号中断，重新来过
        if (errno == EINTR) goto Done;
        result = POLL_ERROR;
        goto Done;
    }
    if (eventCount == 0) {
        // 超时，没有事件
        result = POLL_TIMEOUT;
        goto Done;
    }

    // 遍历就绪的 fd
    for (int i = 0; i < eventCount; i++) {
        int fd = eventItems[i].data.fd;
        uint32_t epollEvents = eventItems[i].events;

        if (fd == mWakeEventFd) {
            if (epollEvents & EPOLLIN) {
                awoken();   // 读走 eventfd 的计数，清零以便下次阻塞
            }
        } else {
            // 处理用户注册的其他 fd（如 vsync fd、input fd）
            // ...
        }
    }

Done:
    // 处理 native Message（C++ 层的 Handler 机制，独立于 Java 层）
    // ...
    return result;
}

// 唤醒：向 eventfd 写 1
void Looper::wake() {
    uint64_t inc = 1;
    // write 是原子的，不需要锁
    // epoll 检测到 mWakeEventFd 可读 → epoll_wait 返回
    ssize_t nWrite = TEMP_FAILURE_RETRY(write(mWakeEventFd, &inc, sizeof(uint64_t)));
}

// 清除唤醒信号
void Looper::awoken() {
    uint64_t counter;
    // 读走 eventfd 的值，将其清零
    // 若不清零，epoll_wait 下次会立即返回（fd 仍可读）
    TEMP_FAILURE_RETRY(read(mWakeEventFd, &counter, sizeof(uint64_t)));
}
```
