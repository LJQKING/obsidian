
## 1. 三者关系图

```
Coroutines(协程框架)
    ├─ suspend 机制 ──┐
    ├─ Scope (GlobalScope, viewModelScope) ──┐
    └─ Dispatcher (Dispatchers.Main/IO) ──┐
                                          │
                                          ↓
                            Flow (冷流：按需生产)
                            ├─ flow {}
                            ├─ 中间操作：map, filter, collect
                            └─ 收集器订阅时才开始执行
                                          │
                                          ↓
                            StateFlow (热流：始终保持值)
                            ├─ 持有最新状态值
                            ├─ 立即发射当前值给新收集器
                            └─ 基于 Flow 实现
```

## 2. 核心概念对比

|特性|Coroutines|Flow|StateFlow|
|---|---|---|---|
|**主要目的**|异步任务管理|异步数据流|状态管理|
|**流冷热**|N/A|冷流（Lazy）|热流（总是活跃）|
|**初值**|N/A|无初值|必有初值|
|**订阅延迟**|N/A|订阅时才开始|订阅时立即发射当前值|
|**线程切换**|Dispatcher|Dispatcher|Dispatcher|
|**内存管理**|生命周期管理|与收集器绑定|需要手动 cancel|

## 3. 详细讲解和源码原理

### 3.1 Coroutines（协程框架基础）

**定义**：Kotlin 协程是轻量级线程，可以被挂起和恢复。

```kotlin
// 基本启动
viewModelScope.launch {
    // 这里是 suspend 上下文
    val result = fetchData()  // suspend function
    updateUI(result)
}

// 返回结果
val data = viewModelScope.async {
    fetchData()
}.await()
```

**Scope 的意义**：

- `GlobalScope`：应用全生命周期（危险，易内存泄漏）
- `viewModelScope`：与 ViewModel 生命周期绑定（Android Jetpack）
- `lifecycleScope`：与 Activity/Fragment 生命周期绑定

**Dispatcher 原理**：

```kotlin
// Dispatcher 决定代码在哪个线程执行
launch(Dispatchers.Main) {        // UI 线程
    updateUI()
}

launch(Dispatchers.IO) {          // IO 线程池
    val data = database.query()
}

launch(Dispatchers.Default) {     // CPU 密集型
    val result = heavyComputation()
}

launch(Dispatchers.Unconfined) {  // 继承父协程的 dispatcher（不推荐）
}
```

**suspend 机制原理**：

```kotlin
suspend fun fetchData(): String {
    // suspend 不是真的暂停线程，而是让协程框架保存状态
    delay(1000)  // 协程暂停，不阻塞线程
    return "data"
}

// 编译后会生成状态机：
// State 0: 调用 delay
// State 1: 获取结果并返回
// 协程框架通过 Continuation 对象传递控制流
```

### 3.2 Flow（冷流：按需生产）

**定义**：Flow 是基于 Coroutines 的**响应式编程**库，提供异步数据流。

#### 创建 Flow

```kotlin
// 方式 1：flow {} 构建器
val myFlow = flow<Int> {
    for (i in 1..3) {
        delay(100)
        emit(i)  // 发射数据
    }
}

// 方式 2：从集合转换
val listFlow = listOf(1, 2, 3).asFlow()

// 方式 3：flowOf（立即发射）
val immediateFlow = flowOf(1, 2, 3)

// 方式 4：从回调转换
val callbackFlow = callbackFlow<String> {
    val listener = OnDataListener { data ->
        trySend(data)  // 发射数据
    }
    api.setListener(listener)
    awaitClose { api.removeListener(listener) }
}
```

#### 订阅和收集

```kotlin
myFlow
    .onStart { println("开始收集") }      // 收集前执行
    .onEach { value -> println(value) }   // 每个值
    .onCompletion { println("完成") }     // 完成或错误
    .collect { value ->
        println("收集到: $value")
    }

// collect 是 suspend function，必须在协程中调用
viewModelScope.launch {
    myFlow.collect { value ->
        // 处理每个值
    }
}
```

#### 中间操作（Operators）

```kotlin
flow {
    emit(1)
    emit(2)
    emit(3)
}
    .map { it * 2 }              // 转换：1→2, 2→4, 3→6
    .filter { it > 2 }            // 过滤：只保留 >2 的值
    .take(2)                      // 只取前 2 个
    .drop(1)                      // 跳过前 1 个
    .distinct()                   // 去重
    .flatMapLatest { id ->        // 切换到新流，取消前一个
        getDetails(id)
    }
    .flatMapConcat { id ->        // 顺序处理多个流
        getDetails(id)
    }
    .buffer(capacity = 10)        // 缓冲区大小
    .conflate()                   // 保留最新值，跳过中间值
    .collectLatest { value ->     // 取消前一个收集，只处理最新
        // 适合搜索场景
    }
```

#### Flow 的冷流特性

```kotlin
val flow = flow {
    println("Flow 开始执行")
    emit(1)
}

// 创建时不执行
println("创建 flow")

// 订阅时才执行
println("订阅 1：")
flow.collect { println(it) }  // 输出："Flow 开始执行"、1

println("订阅 2：")
flow.collect { println(it) }  // 输出："Flow 开始执行"、1（重新执行！）
```

#### Flow 背压处理

```kotlin
// 背压：生产者快于消费者时的处理

// 1. buffer - 缓冲，生产者继续发射
flow {
    repeat(1000) { emit(it) }
}
    .buffer(capacity = 100)  // 缓冲 100 个
    .collect { delay(10) }   // 消费者慢

// 2. conflate - 保留最新，跳过中间值
flow { repeat(1000) { emit(it) } }
    .conflate()              // 只保留最新值
    .collect { delay(10) }

// 3. collectLatest - 取消前一个收集
flow { repeat(1000) { emit(it) } }
    .collectLatest {         // 有新值时取消前一个处理
        delay(10)            // 如果 10ms 内有新值，这个被取消
    }

// 4. 背压异常（默认行为）
flow {
    repeat(1000) { emit(it) }
}
    .collect { delay(10) }   // 会导致 BufferOverflowException
```

### 3.3 StateFlow（热流：始终保持值）

**定义**：StateFlow 是特殊的 Flow，持有一个**可变状态**，总是发射最新值。

#### 创建和使用

```kotlin
// 方式 1：MutableStateFlow
val count = MutableStateFlow(0)

// 方式 2：stateIn 操作符（Flow -> StateFlow）
val uiState: StateFlow<String> = flow {
    emit("状态1")
    delay(1000)
    emit("状态2")
}
    .stateIn(
        scope = viewModelScope,
        started = SharingStarted.Lazily,
        initialValue = "初始值"
    )
```

#### 发射和订阅

```kotlin
val state = MutableStateFlow(0)

// 发射新值
state.value = 1          // 直接赋值
state.emit(2)            // 或 emit（suspend）

// 获取当前值
val current = state.value

// 订阅
viewModelScope.launch {
    state.collect { value ->
        println("状态: $value")
    }
}

// 热流特性：新订阅者立即收到当前值
println("当前值: ${state.value}")  // 输出：当前值: 2
viewModelScope.launch {
    state.collect { println(it) }  // 立即收到 2
}
```

#### 与 Flow 的关键差异

```kotlin
val flow = flow {
    println("Flow 发射 1")
    emit(1)
    delay(1000)
    println("Flow 发射 2")
    emit(2)
}

val state = MutableStateFlow(0)

// Flow 的冷流行为
println("订阅 Flow")
viewModelScope.launch {
    flow.collect { println("Flow: $it") }
}
// 输出：Flow 发射 1、Flow: 1、(等1秒)、Flow 发射 2、Flow: 2

// StateFlow 的热流行为
viewModelScope.launch {
    state.collect { println("State: $it") }
}
// 输出：State: 0（立即收到当前值）
state.value = 1  // 立即发射
state.value = 2  // 立即发射
// 输出：State: 1、State: 2
```

#### SharingStarted 策略

```kotlin
// 控制上游 Flow 何时启动

// 1. SharingStarted.Lazily - 第一个订阅者出现时启动
val state = flow { }
    .stateIn(viewModelScope, SharingStarted.Lazily, "init")
// 优点：不浪费资源；缺点：有延迟

// 2. SharingStarted.Eagerly - 立即启动（订阅前）
val state = flow { }
    .stateIn(viewModelScope, SharingStarted.Eagerly, "init")
// 优点：总是有最新值；缺点：即使没人订阅也在运行

// 3. SharingStarted.WhileSubscribed(5000) - 有订阅者时运行，5s 无订阅后停止
val state = flow { }
    .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), "init")
// 最平衡的方案
```

## 4. 源码原理深度分析

### 4.1 Flow 的冷流实现

```kotlin
// Flow 接口简化版
interface Flow<out T> {
    suspend fun collect(collector: FlowCollector<T>)
}

// flow {} 构建器返回一个新的 Flow 实现类
// 每次 collect 时，都会调用 block 中的代码
// 即使两次 collect，也是两个独立的执行
```

**执行流程**：

```
1. flow { emit(1) } 创建 Flow 对象（不执行）
2. .collect { } 调用时
3. 协程框架创建新的 Continuation
4. 执行 block 中的代码
5. emit(1) 实际上是调用 FlowCollector.emit()
6. 恢复 Continuation，执行 collect 的 lambda
```

### 4.2 StateFlow 的热流实现

```kotlin
// StateFlow 简化源码
class StateFlowImpl<T>(initialValue: T) : StateFlow<T> {
    private var _value = AtomicReference(initialValue)
    
    override val value: T
        get() = _value.get()
    
    override suspend fun collect(collector: FlowCollector<T>) {
        // 1. 立即发射当前值
        collector.emit(_value.get())
        
        // 2. 订阅状态变化
        val subscription = subscribe {
            collector.emit(it)
        }
        
        // 3. 挂起直到取消
        try {
            awaitCancellation()
        } finally {
            subscription.cancel()
        }
    }
}
```

**热流的代价**：

- 内存占用：始终持有一个值
- 无法 GC：被持有的对象不能释放
- 多订阅者共享：所有订阅者看到同一个对象

### 4.3 背压处理的内部机制

```kotlin
// buffer 操作符创建一个通道
fun <T> Flow<T>.buffer(capacity: Int): Flow<T> = flow {
    val channel = Channel<T>(capacity)
    
    // 生产者协程（背景运行）
    launch {
        this@buffer.collect { channel.send(it) }
        channel.close()
    }
    
    // 消费者协程（前景运行）
    for (value in channel) {
        emit(value)
    }
}

// Channel 是一个有界队列
// send() 会挂起直到有空间
// 这样实现了背压
```

## 5. 最佳实践和参数配置

### 5.1 MVVM 中的使用模式

```kotlin
class MyViewModel : ViewModel() {
    // 私有的 MutableStateFlow（可变状态）
    private val _uiState = MutableStateFlow(UiState())
    
    // 公开的 StateFlow（只读状态）
    val uiState: StateFlow<UiState> = _uiState.asStateFlow()
    
    // 私有的 Flow（数据流）
    private val _events = MutableSharedFlow<Event>()
    val events: SharedFlow<Event> = _events.asSharedFlow()
    
    fun loadData() {
        viewModelScope.launch {
            try {
                val data = repository.fetchData()  // suspend
                _uiState.update { it.copy(data = data) }
            } catch (e: Exception) {
                _uiState.update { it.copy(error = e.message) }
            }
        }
    }
}

// UI 层
class MyFragment : Fragment() {
    private val viewModel: MyViewModel by viewModels()
    
    override fun onViewCreated(view: View, savedInstanceState: Bundle?) {
        viewLifecycleOwner.lifecycleScope.launch {
            // repeatOnLifecycle 自动处理生命周期
            repeatOnLifecycle(Lifecycle.State.STARTED) {
                viewModel.uiState.collect { state ->
                    updateUI(state)
                }
            }
        }
    }
}
```

### 5.2 常用参数配置表

```kotlin
// Flow 中间操作参数

// 1. buffer - 背压缓冲
.buffer(capacity = 64)  // 默认 64，可调整 0-128

// 2. conflate - 只保留最新值
.conflate()  // 无参数

// 3. distinctUntilChanged - 去重
.distinctUntilChanged()                    // 完全去重
.distinctUntilChanged { old, new ->        // 自定义去重逻辑
    old.id == new.id  // 返回 true 时跳过发射
}

// 4. debounce - 防抖（等待后发射）
.debounce(timeMillis = 300)  // 300ms 无新值后发射

// 5. throttle - 节流（定时发射）
.throttle(periodMillis = 300)  // 300ms 内只发射一次

// 6. timeout - 超时
.timeout(timeMillis = 5000)  // 5s 无新值则抛异常

// 7. retry - 重试
.retry(retries = 3)            // 重试 3 次
.retry { cause -> cause is IOException }  // 条件重试

// 8. flatMapLatest - 切换流
.flatMapLatest { id ->
    searchRepository.search(id)  // 新搜索自动取消旧搜索
}

// 9. launchIn - 简化订阅（自动管理）
myFlow
    .onEach { updateUI(it) }
    .launchIn(viewModelScope)  // 不需要 collect
```

### 5.3 StateFlow 参数配置

```kotlin
// stateIn 操作符
flow {
    emit("value1")
    delay(1000)
    emit("value2")
}
    .stateIn(
        scope = viewModelScope,           // 必需：协程域
        started = SharingStarted.WhileSubscribed(5000),  // 启动策略
        initialValue = "loading"          // 初值
    )

// SharingStarted 选择指南
// - Eagerly: 数据源启动快，需要始终有值（如计时器）
// - Lazily: 数据源启动慢，节省资源
// - WhileSubscribed(5000): 推荐用于 ViewModel，5s 内无订阅则停止
```

## 6. 常见问题和性能考虑

### 6.1 内存泄漏风险

```kotlin
// ❌ 危险：Global Scope + Flow
GlobalScope.launch {
    myFlow.collect { updateUI(it) }
}
// Flow 可能长期运行，Activity 销毁时不会停止

// ✅ 正确：使用 viewLifecycleOwner
viewLifecycleOwner.lifecycleScope.launch {
    myFlow.collect { updateUI(it) }
}
// 自动随 Fragment 销毁而取消
```

### 6.2 背压导致的 OOM

```kotlin
// ❌ 危险：无限发射
flow {
    repeat(Int.MAX_VALUE) { emit(it) }
}
    .collect { delay(1000) }  // 消费很慢
// 会积压所有值，导致 OOM

// ✅ 解决方案
flow {
    repeat(Int.MAX_VALUE) { emit(it) }
}
    .conflate()  // 只保留最新值
    .collect { delay(1000) }
```

### 6.3 多订阅者问题

```kotlin
// ❌ Flow 重复执行（浪费资源）
val flow = flow { 
    val data = expensiveQuery()  // 数据库查询
    emit(data)
}

// 两个订阅者会查询两次！
flow.collect { /* UI1 */ }
flow.collect { /* UI2 */ }

// ✅ 使用 shareIn 或 StateFlow
val sharedFlow = flow.shareIn(
    scope = viewModelScope,
    started = SharingStarted.Lazily,
    replay = 1  // 保留最后 1 个值给新订阅者
)

sharedFlow.collect { /* UI1 */ }
sharedFlow.collect { /* UI2 */ }  // 复用前一个查询
```

## 7. 总结表格

|方面|Coroutines|Flow|StateFlow|
|---|---|---|---|
|**订阅时机**|launch/async 时|collect 时|立即，发射当前值|
|**重复执行**|每次 launch/async|每次 collect|共享同一个值|
|**适用场景**|异步任务、事件|数据流、序列处理|状态管理、UI 状态|
|**背压处理**|N/A|buffer/conflate/collectLatest|自动处理|
|**内存管理**|通过 Scope|通过 collect 周期|需手动管理|
|**线程安全**|依赖 Dispatcher|依赖 Dispatcher|原子操作|
