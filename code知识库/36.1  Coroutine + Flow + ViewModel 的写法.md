> **MVVM + Repository + Coroutine + StateFlow + Hilt + Retrofit + Room**

```text
UI(Activity/Fragment/Compose)
            │
            ▼
        ViewModel
            │
            ▼
        Repository
         ├── Retrofit
         └── Room
```

# 6. ViewModel标准写法

核心：

```kotlin
MutableStateFlow
```

```kotlin
@HiltViewModel
class UserViewModel @Inject constructor(
    private val repository: UserRepository
) : ViewModel() {

    private val _userState =
        MutableStateFlow<UiState<User>>(UiState.Loading)

    val userState =
        _userState.asStateFlow()

    fun loadUser() {

        viewModelScope.launch {

            try {

                _userState.value =
                    UiState.Loading

                val user =
                    withContext(Dispatchers.IO) {
                        repository.loadUser()
                    }

                _userState.value =
                    UiState.Success(user)

            } catch (e: Exception) {

                _userState.value =
                    UiState.Error(
                        e.message ?: "Unknown Error"
                    )
            }
        }
    }
}
```

# 7. Activity收集Flow

Google推荐：

```kotlin
repeatOnLifecycle()
```

避免页面不可见时继续收集。

```kotlin
@AndroidEntryPoint
class MainActivity : AppCompatActivity() {

    private val viewModel by viewModels<UserViewModel>()

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        lifecycleScope.launch {

            repeatOnLifecycle(
                Lifecycle.State.STARTED
            ) {

                viewModel.userState.collect {

                    when (it) {

                        is UiState.Loading -> {
                            showLoading()
                        }

                        is UiState.Success -> {
                            showUser(it.data)
                        }

                        is UiState.Error -> {
                            showError(it.msg)
                        }
                    }
                }
            }
        }

        viewModel.loadUser()
    }
}
```

# 8. Compose标准写法

Compose中更简单：

```kotlin
@Composable
fun UserScreen(
    vm: UserViewModel = hiltViewModel()
) {

    val state by vm.userState.collectAsState()

    when(state){

        is UiState.Loading -> {
            CircularProgressIndicator()
        }

        is UiState.Success -> {

            Text(
                (state as UiState.Success<User>)
                    .data.name
            )
        }

        is UiState.Error -> {

            Text(
                (state as UiState.Error)
                    .msg
            )
        }
    }
}
```

# 9. Flow分页搜索

例如搜索用户：

```kotlin
private val keyword =
    MutableStateFlow("")
```

输入变化：

```kotlin
fun search(text:String){
    keyword.value = text
}
```

监听：

```kotlin
val usersFlow =
    keyword
        .debounce(500)
        .distinctUntilChanged()
        .flatMapLatest {

            flow {

                emit(
                    repository.search(it)
                )
            }
        }
```

应用场景：

```text
搜索框
实时搜索
自动取消上一次请求
```

类似：

```text

京东
抖音
今日头条
```

搜索功能都这么做。

```kotlin
    _event.asSharedFlow()
```


```kotlin
viewModel.event.collect {

    Toast.makeText(
        this,
        it,
        Toast.LENGTH_SHORT
```

# 企业级 ViewModel 模板

```kotlin
@HiltViewModel
class UserViewModel @Inject constructor(
    private val repository: UserRepository
) : ViewModel() {

    private val _uiState =
        MutableStateFlow<UiState<List<User>>>(
            UiState.Loading
        )

    val uiState =
        _uiState.asStateFlow()

    fun getUsers() {

        viewModelScope.launch {

            runCatching {

                repository.getUsers()

            }.onSuccess {

                _uiState.value =
                    UiState.Success(it)

            }.onFailure {

                _uiState.value =
                    UiState.Error(
                        it.message ?: ""
                    )
            }
        }
    }
}
```

这套 **Coroutine + StateFlow + SharedFlow + ViewModel + Repository** 的写法基本就是 2025~2026 年 Android 大厂（字节、腾讯、阿里、美团）MVVM 项目的主流标准方案。相比 LiveData，StateFlow/SharedFlow 在协程体系内更统一、功能更强，也更适合 Compose。
